map.fromLatLngToPoint()

    Drag the marker on the map, show the pixels from left-top position. 

    var div = document.getElementById("map_canvas");
var map = plugin.google.maps.Map.getMap(div);
map.one(plugin.google.maps.event.MAP_READY, function() {

  // From v1.2.0, you can get the camera center position without callback.
  var center = map.getCameraTarget();

  map.addMarker({
    position: center,
    draggable: true,
    title: "Drag me!"
  }, function(marker) {
    marker.showInfoWindow();

    // Catch the marker drag event
    marker.on(plugin.google.maps.event.MARKER_DRAG_END, function(latLng) {
        // LatLng -> Point
        map.fromLatLngToPoint(latLng, function(point) {

          // Show on the marker
          marker
            .setTitle(
              "left : " + point[0].toFixed(1) + "px\n" + "top : " + point[1].toFixed(1) + "px"
            )
            .showInfoWindow();
        });

    });

  });
});
